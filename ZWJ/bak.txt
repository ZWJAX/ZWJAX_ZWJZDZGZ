<script setup>
import { ref, computed, watchEffect, reactive } from 'vue';

// ÁôªÂΩïÁõ∏ÂÖ≥Áä∂ÊÄÅ
const isLogin = ref(false); // ÊòØÂê¶Â∑≤ÁôªÂΩï
const showPwd = ref(false); // ÂØÜÁ†ÅÊòØÂê¶ÂèØËßÅ
const loginForm = reactive({
  username: '',
  password: ''
});
// ÁôªÂΩïÂ§ÑÁêÜ
function handleLogin() {
  if (loginForm.username && loginForm.password) {
    isLogin.value = true;
  } else {
    alert('ËØ∑ËæìÂÖ•Áî®Êà∑ÂêçÂíåÂØÜÁ†Å');
  }
}
// ÈÄÄÂá∫ÁôªÂΩï
function logout() {
  isLogin.value = false;
  loginForm.username = '';
  loginForm.password = '';
}

// ‰∏≠ÊñáÂàÜÂå∫Êï∞ÊçÆ
const categories = [
  { name: 'Âä®Êº´', icon: 'img', color: '#a78bfa' },
  { name: 'ÂΩ±ËßÜ', icon: 'üé•', color: '#fbbf24' },
  { name: 'Ê∏∏Êàè', icon: 'üéÆ', color: '#f472b6' },
  { name: 'ÁîüÊ¥ª', icon: 'üåà', color: '#34d399' },
  { name: 'Êï∞Á†Å', icon: 'üíª', color: '#60a5fa' },
  { name: 'ËΩ¶', icon: 'üöó', color: '#7c3aed' },
  { name: 'Êàø', icon: 'üè†', color: '#ffb700' }
];
const selected = ref('Âä®Êº´'); // ÂΩìÂâçÈÄâ‰∏≠ÁöÑÂàÜÂå∫
const navActive = ref('È¶ñÈ°µ'); // ÂΩìÂâçÂØºËà™ÊøÄÊ¥ªÈ°π
const posts = ref([]); // Â∏ñÂ≠êÂàóË°®
const showForm = ref(false); // ÊòØÂê¶ÊòæÁ§∫ÂèëÂ∏ñÂºπÁ™ó
const loading = ref(false); // Âä†ËΩΩÂä®Áîª
const search = ref(''); // ÊêúÁ¥¢ÂÖ≥ÈîÆÂ≠ó
const form = ref({
  category: '',
  title: '',
  content: '',
  author: '',
  video: ''
}); // ÂèëÂ∏ñË°®Âçï
const commentInputs = reactive({}); // ËØÑËÆ∫ËæìÂÖ•Ê°ÜÂÜÖÂÆπ

// ÁõëÂê¨Â∏ñÂ≠êÂèòÂåñÔºåÂàùÂßãÂåñËØÑËÆ∫ËæìÂÖ•Ê°Ü
watchEffect(() => {
  posts.value.forEach(post => {
    if (!commentInputs[post.id]) {
      commentInputs[post.id] = { content: '' };
    }
  });
});

// Ëé∑ÂèñÂàÜÂå∫È¢úËâ≤
function categoryColor(catName) {
  const cat = categories.find(c => c.name === catName);
  return cat ? cat.color : '#a78bfa';
}

// Ëé∑ÂèñÂ∏ñÂ≠ê
async function fetchPosts() {
  loading.value = true;
  let url = '';
  if (navActive.value === 'È¶ñÈ°µ') {
    url = `http://localhost:3001/api/posts?category=${selected.value}`;
  } else {
    url = `http://localhost:3001/api/posts`;
  }
  try {
    const res = await fetch(url);
    posts.value = await res.json();
  } catch (e) {
    posts.value = [];
    alert('Êó†Ê≥ïËøûÊé•ÂêéÁ´ØÊúçÂä°ÔºåËØ∑Ê£ÄÊü•Node.jsÂêéÁ´ØÊòØÂê¶ÂêØÂä®ÔºÅ');
  } finally {
    setTimeout(() => { loading.value = false; }, 500);
  }
}

// ÂàáÊç¢ÂàÜÂå∫
function changeCategory(cat) {
  selected.value = cat;
  navActive.value = 'È¶ñÈ°µ';
  fetchPosts();
}

// ÊêúÁ¥¢ËøáÊª§
const filteredPosts = computed(() => {
  if (!search.value) return posts.value;
  return posts.value.filter(
    p =>
      p.title.includes(search.value) ||
      p.content.includes(search.value) ||
      p.author.includes(search.value)
  );
});

// ÁõëÂê¨ÂØºËà™ÊàñÂàÜÂå∫ÂèòÂåñËá™Âä®Ëé∑ÂèñÂ∏ñÂ≠ê
watchEffect(fetchPosts);

// ÂèëÂ∏ñ
async function submitPost() {
  if (!form.value.category || !form.value.title || !form.value.content || !form.value.author) return;
  loading.value = true;
  await fetch('http://localhost:3001/api/posts', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(form.value)
  });
  showForm.value = false;
  form.value = { category: '', title: '', content: '', author: '', video: '' };
  await fetchPosts();
  loading.value = false;
}

// ËØÑËÆ∫
async function submitComment(post) {
  const input = commentInputs[post.id];
  if (!input.content) return;
  await fetch(`http://localhost:3001/api/posts/${post.id}/comments`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      content: input.content,
      time: new Date().toLocaleString()
    })
  });
  input.content = '';
  await fetchPosts();
}
</script>
